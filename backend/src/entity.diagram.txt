This is the basic schema for now, can change later , according to need.
users [icon: user, color: blue] {
  id string [pk]
  name string
  email string [unique]
  username string
  password string
  profileImageUrl string // from cloudinary  
  coverImageUrl string, // from cloudinary
  createdAt Date
  updatedAt Date
  lastLogin Date
  DeletedAt Date
}
user_sessions [icon: clock, color: purple] {
  id string pk
  sessionToken string unique
  userId string fk nullable [ref: > users.id] 
  createdAt timestamp
  expiresAt timestamp
  lastActivity timestamp
}

expense[icon : money, color : green]{
  id string [pk]
  userId string [fk] [ref: > users.id]// ref to users.id
  category string  // or enum, but nah , string is good
  description string
  amount int  // or maybe string, but nah , int is good 
  createdAt Date
  updatedAt Date
  DeletedAt Date
}
